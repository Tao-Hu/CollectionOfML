Comparison of the LASSO and adaptive LASSO estimators

Simulate a fake dataset

 set.seed(0820)

 n         <- 50
 p         <- 25
 beta      <- rep(0,p)
 beta[1:5] <- 1:5/5

 X <- matrix(rnorm(n*p),n,p)
 X <- scale(X)

 Xb <- X%*%beta
 Y <- X%*%beta+rnorm(n)
 Y <- Y-mean(Y)

 library(lars)

Let's repeat this many times in a full simulation study

This code generates 100 datasets and computes the LASSO and A-LASSO solution for each dataset. We can then compare properties of the sampling distributions, such as bias, variance and mean squared error (MSE).

 nsims <- 100
 beta1 <- matrix(0,nsims,p)
 beta2 <- matrix(0,nsims,p)

 for(sim in 1:nsims){

   X <- matrix(rnorm(n*p),n,p)
   X <- scale(X) 

   Xb <- X%*%beta
   Y <- X%*%beta+rnorm(n)
   Y <- Y-mean(Y)


   # Fit the lasso

    lasso      <- lars(X,Y)
    df         <- lasso$df
    MSE        <- lasso$RSS/n
    bic        <- log(n)*df+n*log(MSE)
    beta_lasso <- lasso$beta[which.min(bic),] 

   # Fit the adaptive lasso

    beta_ols <- lm(Y~X)$coef[-1]
    w        <- abs(beta_ols)
    Xw       <- scale(X,center=FALSE,scale=1/w)
    alasso   <- lars(Xw,Y,normalize=FALSE)
    df          <- alasso$df
    MSE         <- alasso$RSS/n
    bic         <- log(n)*df+n*log(MSE)
    beta_alasso <- w*alasso$beta[which.min(bic),] 

    beta1[sim,]<-beta_lasso
    beta2[sim,]<-beta_alasso
 }

Plot the sampling distribution (boxplots) versus the true values (red line)

 boxplot(beta1,main="LASSO")
 lines(beta,col=2)


 boxplot(beta2,main="A-LASSO")
 lines(beta,col=2)

Compute bias, variance and MSE

 beta  <- matrix(beta,nsims,p,byrow=TRUE)

 bias1 <- colMeans(beta1-beta)
 bias2 <- colMeans(beta2-beta)

 var1  <- apply(beta1,2,var)
 var2  <- apply(beta2,2,var)

 mse1  <- colMeans((beta1-beta)^2)
 mse2  <- colMeans((beta2-beta)^2)

 plot(bias1,type="l",xlab="Variable",ylab="Bias")
 lines(bias2,col=2)
 legend("bottomright",c("Lasso","A-Lasso"),lty=1,col=1:2,inset=0.05)


 plot(var2,col=2,type="l",xlab="Variable",ylab="Variance")
 lines(var1,col=1)
