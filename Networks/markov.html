Network prediction using a Gaussian Markov random field

Load a picture of Lena. The nodes are the \(256^2\) pixels and the attribute is the color

 opts_chunk$set(out.width='900px', dpi=200)

 library(fields)
 library(png)

 x <- readPNG("S:\\Desktop\\lena.png")
 dim(x)

## [1] 256 256

 x <- t(x[256:1,]) 

 col <- gray(seq(0,1,.01))
 image.plot(x,col=col,main="Full original image")

Zoom in on her face

 x <- x[100:200,50:150]
 image.plot(x,col=col,main="Lena")

Pretend some attributes are missing

  s1 <- sample(2:100,10,replace=TRUE)
  s2 <- sample(2:100,10,replace=TRUE)

  X      <- x
  X[s1,] <- NA
  X[,s2] <- NA
  test   <- is.na(X)

  image.plot(X,col=col,main="Thinned image")

Adjacency and Laplacian matrix

  X <- as.vector(X)
  s <- as.matrix(expand.grid(1:101,1:101))
  s[1:10,]

##       Var1 Var2

  A <- ifelse(rdist(s,s)==1,1,0)
  m <- rowSums(A)

  L <- -A
  diag(L)<-m

Prediction

  xbar <- mean(X[!test])
  Lpp  <- L[test,test]
  Lpo  <- L[test,!test]

  Xhat <- X
  Xhat[test] <- xbar - solve(Lpp)%*%Lpo%*%(X[!test]-xbar)
  image.plot(matrix(Xhat,101,101),col=col,main="Reconstructed image")
